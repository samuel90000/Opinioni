class Libro:
    def __init__(self, titolo, autore, anno):
        self.titolo = titolo
        self.autore = autore
        self.anno = anno

    def get_titolo(self):
        return self.titolo

    def set_titolo(self, titolo):
        self.titolo = titolo

    def toString(self):
        return "autore: " + self.autore + ", anno: " + str(self.anno)

libro = Libro("benito mussolini", "mussolini", "1945")
print(libro.toString())
print(libro.get_titolo())
